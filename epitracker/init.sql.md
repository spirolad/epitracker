-- 1. TABLE DES PROFILS
CREATE TABLE IF NOT EXISTS profiles (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  email TEXT UNIQUE NOT NULL,
  full_name TEXT,
  temp_password TEXT,
  role TEXT CHECK (role IN ('admin', 'apprenant')) DEFAULT 'apprenant',
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now())
);

CREATE TABLE courses (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  name TEXT NOT NULL,
  latitude DOUBLE PRECISION NOT NULL,
  longitude DOUBLE PRECISION NOT NULL,
  radius_meters INTEGER DEFAULT 50,
  start_at TIMESTAMP WITH TIME ZONE NOT NULL, -- Horaire de début
  end_at TIMESTAMP WITH TIME ZONE NOT NULL,   -- Horaire de fin
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now())
);

-- On recrée la table des présences
CREATE TABLE attendances (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  course_id bigint REFERENCES courses(id) ON DELETE CASCADE,
  student_id bigint REFERENCES profiles(id) ON DELETE CASCADE,
  scanned_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()),
  latitude DOUBLE PRECISION,
  longitude DOUBLE PRECISION,
  is_valid BOOLEAN DEFAULT false,
  CONSTRAINT unique_student_course UNIQUE(course_id, student_id)
);

-- 4. DÉSACTIVATION SÉCURITÉ
ALTER TABLE profiles DISABLE ROW LEVEL SECURITY;
ALTER TABLE courses DISABLE ROW LEVEL SECURITY;
ALTER TABLE attendances DISABLE ROW LEVEL SECURITY;
